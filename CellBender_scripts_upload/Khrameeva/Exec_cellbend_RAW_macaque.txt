#!/bin/bash -l

#SBATCH -N 1 --ntasks-per-node=6 --partition=gpu
#SBATCH --gres=gpu:1
#SBATCH --job-name="PAIR"
#SBATCH --output=/home/dennis00/Yi_lab/CellBender_scripts/slurm-archive/%j.out

module load cuda/11.7

cd $SLURM_SUBMIT_DIR

cd /home/dennis00/Yi_lab/CellBender_scripts/Khrameeva

cellbender remove-background \
                  	--input "/home/dennis00/Yi_lab/raw_h5_data/Khrameeva_h5/Macaque.h5" \
                  	--output "/home/dennis00/Yi_lab/cellbent_output_RAW/Khrameeva/Macaque.h5" \
                  	--cuda \
			--expected-cells 17000 \
                        --total-droplets-included 20000 \
                        --low-count-threshold 15 \
                  	--learning-rate 1e-4 \
                  	--fpr 0.01 \
                 	--epochs 150

echo cellbender_Khrameeva_Macaque
echo "SLURM Job ID        : ${SLURM_JOB_ID}"
