#!/bin/bash -l

#SBATCH -N 1 --ntasks-per-node=6 --partition=gpu
#SBATCH --gres=gpu:1
#SBATCH --job-name="PAIR"
#SBATCH --output=/home/dennis00/Yi_lab/CellBender_scripts/slurm-archive/%j.out

module load cuda/11.7

cd $SLURM_SUBMIT_DIR

cd /home/dennis00/Yi_lab/CellBender_scripts/Kanton

cellbender remove-background \
                  	--input "/home/dennis00/Yi_lab/raw_h5_data/Kanton_h5/set_1.h5" \
                  	--output "/home/dennis00/Yi_lab/cellbent_output_RAW/Kanton/set_1.h5" \
                  	--cuda \
                   	--expected-cells 10000 \
                  	--total-droplets-included 25000 \
                 	--low-count-threshold 5 \
                  	--learning-rate 1e-4 \
                  	--fpr 0.01 \
                 	--epochs 20

echo cellbender_Kanton_set_1
echo "SLURM Job ID        : ${SLURM_JOB_ID}"
